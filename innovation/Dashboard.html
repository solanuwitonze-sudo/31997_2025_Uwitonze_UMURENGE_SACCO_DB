-- This script generates this dashboard dynamically from the Oracle Database
-- Innovation: PL/SQL generates HTML for real-time reporting

CREATE OR REPLACE PROCEDURE generate_dashboard IS
    v_total_savings NUMBER;
    v_total_loans NUMBER;
    v_overdue_loans NUMBER;
    CURSOR c_members IS
        SELECT full_name, member_id FROM members WHERE status = 'ACTIVE';
BEGIN
    -- Query live data
    SELECT NVL(SUM(amount), 0) INTO v_total_savings FROM savings_deposits;
    SELECT NVL(SUM(principal_amount), 0) INTO v_total_loans FROM loans;
    SELECT COUNT(*) INTO v_overdue_loans FROM loans 
    WHERE loan_status = 'ACTIVE' AND due_date < SYSDATE;

    -- Generate HTML with live data
    DBMS_OUTPUT.PUT_LINE('Total Savings: ' || v_total_savings);
    DBMS_OUTPUT.PUT_LINE('Total Loans: ' || v_total_loans);
    DBMS_OUTPUT.PUT_LINE('Overdue Loans: ' || v_overdue_loans);
END;
